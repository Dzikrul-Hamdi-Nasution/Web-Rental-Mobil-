  });

    nilai:nilai_id,
  });
  firebase.database().ref("user").child(pecah_id).child("mobil_last").set({

  });

    nilai:nama,
  });

    waktu: pecah,
    harga:tampil_harga,
    service:pelayanan,
  });

}
